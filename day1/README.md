## Переменные
```

var isWeekend = true; // булева переменная (boolean)

var noValue; // а у этой переменной есть только имя, но нет значения (undefined)

var a = 1, b = 2, c = a + b; // объявление сразу нескольких переменных одним var'ом
```

# Вывод в консоль
```
console.log(1000); // выводим число

console.log('Moscow Coding School'); // выводим строку

console.log(myName); // выводим переменную

console.log(typeof currentYear); // выводим тип переменной (number)
```

# Математические операторы
```
// сложение
var week = 5 + 2; // 7

// вычитание
console.log(999 - 777); // 222

// умножение
var month = week * 4; // 28

// деление
console.log(month / 7); // 4

// целочисленный остаток от деления
console.log(month % 2); // 0
console.log(month % 3); // 1
```

# Операторы присвоения
```
// объявление переменной и присвоение ей значения
var a = 1;

// прибавление, умножение и деление текущего значения переменной
// и запись результата в переменную
a += 5; // a = a + 5
a -= 3; // a = a -3
a *= 2; // a = a * 2
a /= 4; // a = a / 4

// Инкремент (++) и декремент (a--)
a = 1;
a++; // a = a + 1
a--; // a = a - 1
++a; // a = a + 1
--a; // a = a - 1
```

# Условия (if)
```
if (a == 1) {
    console.log('условие выполнилось');
}

if (a == 2) {
    // код...
    // выполнится в случае, когда a = 2
} else {
    // код...
    // выполнится во всех других случаях
}

if (a == 1) {
    // когда а равно 1
} else if (a > 2 && a < 10) {
    // когда а больше 2 и меньше 10
} else if (a >= 10 || a == 0) {
    // когда a больше или равно 10
    // или когда a равно 0
} else {
    // во всех остальных случаях
}
```

## Оператор switch

## Циклы
```

```

## Массивы
```
// Массив из 4 элементов-строк с индексами 0, 1, 2 ,3
var names = ['Александр', 'Мария', 'Владимир', 'Ольга'];

console.log(names[0]); // Александр
console.log(names[1]); // Ольга
console.log(names[4]); // undefined (элемента с таким индексом в массиве нет)

// При помощи цикла for выводим в консоль все элементы массива
for (i = 0; i < names.length; i++) {
    console.log(names[i]);
}

// Массив из 6 элементов-чисел
var numbers = [2, 3, 8, 10, -4, 0];

// При помощи цикла for выводим в консоль каждый элемент массива и его индекс
for (i = 0; i < numbers.length; i++) {
    console.log('Элемент: ' + numbers[i] + ', индекс: ' + i);
}

// При помощи цикла for выводим в консоль только четные элементы массива
for (i = 0; i < numbers.length; i++) {
    if (numbers[i] % 2 == 0) {
        console.log(numbers[i] + ' — четное число');
    }
}
```
